-- // HELPER MODULES //
local function AddCorner(obj, radius)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, radius or 4)
    c.Parent = obj
    return c
end

local function AddStroke(obj, color, thickness)
    local s = Instance.new("UIStroke")
    s.Color = color or Style.FrameBorder
    s.Thickness = thickness or 1
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    s.Parent = obj
    return s
end

local function PlayTween(obj, info, target)
    local t = TweenService:Create(obj, info, target)
    t:Play()
    return t
end

    ActiveTab = tabName

    local btn = Instance.new("TextButton", TabBar)

local function CreateSlider(parent, name, min, max, configKey)
    local row = Instance.new("Frame", parent); row.Size = UDim2.new(1, 0, 0, 26); row.BackgroundTransparency = 1
    local lbl = Instance.new("TextLabel", row); lbl.Size = UDim2.new(0.4, 0, 1, 0); lbl.Position = UDim2.new(0, 6, 0, 0); lbl.Text = name; lbl.Font = Style.Font; lbl.TextSize = Style.FontSize; lbl.TextColor3 = Style.Text; lbl.TextXAlignment = Enum.TextXAlignment.Left; lbl.BackgroundTransparency = 1
    
    local bar = Instance.new("TextButton", row); bar.Size = UDim2.new(0.55, -6, 0, 10); bar.Position = UDim2.new(0.45, 0, 0.5, -5); bar.BackgroundColor3 = Style.FrameBg; bar.BorderSizePixel = 0
    AddCorner(bar, 3)
    AddStroke(bar, Style.FrameBorder, 1)
    
    local fill = Instance.new("Frame", bar); fill.Size = UDim2.new((Settings[configKey]-min)/(max-min), 0, 1, 0); fill.BackgroundColor3 = Style.Accent; fill.BorderSizePixel = 0
    AddCorner(fill, 3)
    
    local valLbl = Instance.new("TextLabel", bar); valLbl.Size = UDim2.new(1, 0, 1, 0); valLbl.Text = tostring(Settings[configKey]); valLbl.Font = Style.Font; valLbl.TextSize = Style.FontSize - 2; valLbl.TextColor3 = Style.Text; valLbl.BackgroundTransparency = 1
    
    bar.MouseButton1Down:Connect(function()
        local c; c = RunService.RenderStepped:Connect(function()
            local x = math.clamp((UserInputService:GetMouseLocation().X - bar.AbsolutePosition.X) / bar.AbsoluteSize.X, 0, 1)
            local val = min + (max - min) * x; val = max > 10 and math.floor(val) or math.floor(val*100)/100
            Settings[configKey] = val; fill.Size = UDim2.new(x, 0, 1, 0); valLbl.Text = tostring(val)
            if not UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then c:Disconnect() end
        end)
    end)
end

    local btn = Instance.new("TextButton", row); btn.Size = UDim2.new(0.55, -6, 0, 20); btn.Position = UDim2.new(0.45, 0, 0.5, -10); btn.BackgroundColor3 = Style.FrameBg; btn.Font = Style.Font; btn.TextSize = Style.FontSize; btn.TextColor3 = Style.Accent
    AddCorner(btn, 3)
    AddStroke(btn, Style.FrameBorder, 1)
    
    btn.MouseButton1Click:Connect(function()
        local currentIdx = table.find(options, Settings[configKey]) or 1
        local nextIdx = currentIdx + 1

        Settings[configKey] = options[nextIdx]
        btn.Text = tostring(options[nextIdx])
    end)
    btn.Text = tostring(Settings[configKey])
end

-- // STABILITY RESET MODULE //
local function CleanESP(p)
    if ESPObjects[p] then
        for _, obj in pairs(ESPObjects[p]) do obj.Visible = false end
    end
    if SkeletonObjects[p] then
        for _, line in pairs(SkeletonObjects[p]) do line.Visible = false end
    end
end

-- // MATHEMATICAL LOGIC OPERATIONS //
local function IsVisible(targetPart, checkType)
    if not Settings[checkType] then return true end
    local origin = Camera.CFrame.Position
    local dir = (targetPart.Position - origin)
    local params = RaycastParams.new(); params.FilterType = Enum.RaycastFilterType.Exclude; params.FilterDescendantsInstances = {LocalPlayer.Character, Camera}
    local hit = workspace:Raycast(origin, dir, params)
    return not hit or hit.Instance:IsDescendantOf(targetPart.Parent)
end

local function GetClosest()
    if Settings.StickyAim and CurrentTarget and CurrentTarget.Parent then
        local hum = CurrentTarget.Parent:FindFirstChild("Humanoid")
        if hum and hum.Health > 0 and IsVisible(CurrentTarget, "WallCheckAim") then
            local pos, onScreen = Camera:WorldToViewportPoint(CurrentTarget.Position)
            if onScreen then return CurrentTarget end
        end
    end
    local target, minDst = nil, Settings.FOV
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
            if Settings.TeamCheckAim and p.Team == LocalPlayer.Team then continue end
            local part = p.Character:FindFirstChild(Settings.AimPart)
            if part and IsVisible(part, "WallCheckAim") then
                local pos, onScreen = Camera:WorldToViewportPoint(part.Position)
                local dist = (Vector2.new(pos.X, pos.Y) - UserInputService:GetMouseLocation()).Magnitude
                if onScreen and dist < minDst then minDst = dist; target = part end
            end
        end
    end
    return target
end

-- // REALTIME ENVIRONMENT CALCULATIONS //
RunService.RenderStepped:Connect(function(deltaTime)
    WatermarkObj.Visible = Settings.Watermark
    if Settings.Watermark then WatermarkObj.Text = "Premium Suite v9.0 | Client Mode: " .. ChosenModule end

    PerfList.Visible = Settings.ShowPerfStats
    if Settings.ShowPerfStats then
        PerfList.Text = string.format("FPS: %d\nPING: %d ms\nMEM: %d MB", math.floor(1/deltaTime), math.floor(Stats.Network.ServerLookback:GetValue()), math.floor(Stats:GetTotalMemoryUsageMb()))
    end

    -- Server Tab Real-time Data Hydration Loop
    if MainFrame.Visible and ActiveTab == "Server" then
        -- Count population
        DisplayCount.Text = string.format("%d / %d Players", #Players:GetPlayers(), Players.MaxPlayers)
        
        -- Compute running uptime clock
        local timeInSecs = workspace.DistributedGameTime
        local hours = math.floor(timeInSecs / 3600)
        local mins = math.floor((timeInSecs % 3600) / 60)
        local secs = math.floor(timeInSecs % 60)
        DisplayUptime.Text = string.format("%02d:%02d:%02d", hours, mins, secs)
        
        -- Approximate regional nodes via ping response thresholds
        local ping = Stats.Network.ServerLookback:GetValue()
        if ping < 45 then DisplayRegion.Text = "US East Low-Latency Node"
        elseif ping >= 45 and ping < 100 then DisplayRegion.Text = "US West Main Backbone"
        elseif ping >= 100 and ping < 180 then DisplayRegion.Text = "EU Central Cluster"
        else DisplayRegion.Text = "APAC / Oceanic Relay Node" end
    end

    -- Real-Time Black Ball Simulation Engine Loop
    if MainFrame.Visible then
        for _, flake in pairs(Snowflakes) do
            local currentPos = flake.Object.Position
            local newY = currentPos.Y.Offset + (flake.Speed * deltaTime)
            local newX = currentPos.X.Scale + ((flake.Drift / 1000) * deltaTime)
            
            if newY > Camera.ViewportSize.Y or newX < 0 or newX > 1 then
                flake.Object.Position = UDim2.new(math.random(), 0, 0, -20)
                flake.Speed = math.random(40, 95)
            else
                flake.Object.Position = UDim2.new(newX, 0, 0, newY)
            end
        end
    end

    -- Dynamic WalkSpeed Hook Modification
    if Settings.WalkSpeedMod and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
        LocalPlayer.Character.Humanoid.WalkSpeed = Settings.WalkSpeedVal
    end

    local activeFeatures = {"Active Modules [" .. ChosenModule .. "]"}
    if Settings.Aimbot then table.insert(activeFeatures, "- Engine Aimbot (" .. Settings.AimPart .. ")") end
    if Settings.ESPSkeleton then table.insert(activeFeatures, "- Bone Matrix System") end
    if Settings.WalkSpeedMod then table.insert(activeFeatures, "- Speed Configuration Active") end
    HUDList.Text = table.concat(activeFeatures, "\n")

    local center = Camera.ViewportSize / 2
    local crosshairVisible = Settings.CustomCrosshair and MainFrame.Visible == false 
    for _, l in pairs(CrosshairLines) do l.Visible = crosshairVisible; l.Color = Palette[Settings.CrosshairCol] end
    
    if crosshairVisible then
        local gap, size = 4, 10
        CrosshairLines.HLeft.From = Vector2.new(center.X - gap - size, center.Y)
        CrosshairLines.HLeft.To = Vector2.new(center.X - gap, center.Y)
        CrosshairLines.HRight.From = Vector2.new(center.X + gap, center.Y)
        CrosshairLines.HRight.To = Vector2.new(center.X + gap + size, center.Y)
        CrosshairLines.VTop.From = Vector2.new(center.X, center.Y - gap - size)
        CrosshairLines.VTop.To = Vector2.new(center.X, center.Y - gap)
        CrosshairLines.VBottom.From = Vector2.new(center.X, center.Y + gap)
        CrosshairLines.VBottom.To = Vector2.new(center.X, center.Y + gap + size)
    end

    FOVCircle.Visible = Settings.DrawFOV
    if Settings.DrawFOV then
        FOVCircle.Position = UserInputService:GetMouseLocation()
        FOVCircle.Radius = Settings.FOV
        FOVCircle.Color = Palette[Settings.FOVCol]
    end

    -- Loop Validation ESP Optimization
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer then
            local char = p.Character
            if char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Humanoid") and char.Humanoid.Health > 0 then
                local root = char.HumanoidRootPart
                
                if (Settings.ESPTeamCheck and p.Team == LocalPlayer.Team) or not IsVisible(root, "ESPWallCheck") then
                    CleanESP(p)
                    continue
                end

                if not ESPObjects[p] then
                    ESPObjects[p] = {Box = Drawing.new("Square"), Bar = Drawing.new("Line"), Name = Drawing.new("Text"), Dist = Drawing.new("Text")}
                    ESPObjects[p].Name.Center = true; ESPObjects[p].Dist.Center = true
                    ESPObjects[p].Name.Outline = true; ESPObjects[p].Dist.Outline = true
                end

                local top, onScreen1 = Camera:WorldToViewportPoint(root.Position + Vector3.new(0, 2.3, 0))
                local bottom, onScreen2 = Camera:WorldToViewportPoint(root.Position - Vector3.new(0, 3, 0))

                if onScreen1 or onScreen2 then
                    local height = math.abs(top.Y - bottom.Y)
                    local width = height / 1.8
                    local obj = ESPObjects[p]
                    
                    obj.Box.Visible = Settings.ESPBox
                    obj.Box.Size = Vector2.new(width, height)
                    obj.Box.Position = Vector2.new(top.X - width/2, top.Y)
                    obj.Box.Color = Palette[Settings.BoxCol]
                    
                    obj.Bar.Visible = Settings.ESPHealth
                    obj.Bar.From = Vector2.new(top.X - width/2 - 5, top.Y + height)
                    obj.Bar.To = Vector2.new(top.X - width/2 - 5, top.Y + height - (height * (char.Humanoid.Health / char.Humanoid.MaxHealth)))
                    obj.Bar.Color = Palette[Settings.HealthCol]
                    
                    obj.Name.Visible = Settings.ESPNames
                    obj.Name.Position = Vector2.new(top.X, top.Y - 16)
                    obj.Name.Text = p.Name
                    obj.Name.Color = Palette[Settings.NameCol]
                    obj.Name.Size = 13
                    
                    obj.Dist.Visible = Settings.ESPDistance
                    obj.Dist.Position = Vector2.new(bottom.X, bottom.Y + 2)
                    obj.Dist.Text = math.floor((Camera.CFrame.Position - root.Position).Magnitude) .. "m"
                    obj.Dist.Color = Palette[Settings.DistCol]
                    obj.Dist.Size = 12
                else
                    CleanESP(p)
                end

                -- Structural Skeleton Controller
                if not SkeletonObjects[p] then SkeletonObjects[p] = {} end
                local skeletonSetup = char.Humanoid.RigType == Enum.HumanoidRigType.R15 and R15Skeleton or R6Skeleton
                
                for i, bonePair in ipairs(skeletonSetup) do
                    if not SkeletonObjects[p][i] then
                        SkeletonObjects[p][i] = Drawing.new("Line")
                        SkeletonObjects[p][i].Thickness = 1.2
                    end
                    
                    local line = SkeletonObjects[p][i]
                    local b1, b2 = char:FindFirstChild(bonePair[1]), char:FindFirstChild(bonePair[2])
                    
                    if b1 and b2 and Settings.ESPSkeleton and (onScreen1 or onScreen2) then
                        local p1, o1 = Camera:WorldToViewportPoint(b1.Position)
                        local p2, o2 = Camera:WorldToViewportPoint(b2.Position)
                        
                        if o1 and o2 then
                            line.From = Vector2.new(p1.X, p1.Y)
                            line.To = Vector2.new(p2.X, p2.Y)
                            line.Color = Palette[Settings.BoxCol]
                            line.Visible = true
                        else line.Visible = false end
                    else line.Visible = false end
                end
            else
                CleanESP(p)
            end
        end
    end

    if Settings.Aimbot and Aiming then
        CurrentTarget = GetClosest()
        if CurrentTarget and math.random(1, 100) <= Settings.HitChance then
            local pos, onScreen = Camera:WorldToViewportPoint(CurrentTarget.Position)
            if onScreen then
                local mouseLoc = UserInputService:GetMouseLocation()
                local diff = Vector2.new(pos.X - mouseLoc.X, pos.Y - mouseLoc.Y)
                if diff.Magnitude > Settings.Deadzone then 
                    mousemoverel(diff.X * Settings.Smoothness, diff.Y * Settings.Smoothness) 
                end
            end
        end
    else CurrentTarget = nil end
end)
